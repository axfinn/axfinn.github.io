    todoIng 团队协作与权限管理深度解析

    在前面的文章中，我们已经介绍了 todoIng 的核心模块和 Git 风格的历史追踪系统。今天我们来探讨 todoIng 的团队协作功能和权限管理系统，这是实现高效团队任务管理的关键。

    团队协作的重要性

    在现代工作中，很少有任务是完全独立完成的。大多数项目都需要团队成员之间的协作，这就要求任务管理系统具备强大的团队协作功能：

    1. 任务分配：能够将任务分配给特定的团队成员
    2. 进度共享：团队成员可以实时查看任务进度
    3. 沟通协作：在任务上下文中进行讨论和沟通
    4. 权限控制：确保每个成员只能访问和操作自己有权限的内容

    系统设计思路

    todoIng 的团队协作系统采用了基于角色的访问控制（RBAC）模型，并结合了项目组的概念来组织团队成员和任务。

    权限管理系统实现

    todoIng 的权限管理系统采用了多层权限控制机制，确保系统的安全性和灵活性。

    权限层级

    1. 系统级权限：控制用户能否创建团队、项目等
    2. 团队级权限：控制用户在团队中的操作权限
    3. 项目级权限：控制用户在项目中的操作权限
    4. 任务级权限：控制用户对特定任务的操作权限

    权限检查中间件

    // backend/src/middleware/permission.js
const Team = require('../models/Team');
const Project = require('../models/Project');
const Task = require('../models/Task');

// 检查团队权限
const checkTeamPermission = (requiredPermission) => {
  return async (req, res, next) => {
    try {
      const { teamId } = req.params;
      const userId = req.user._id;
      
      const team = await Team.findById(teamId);
      if (!team) {
        return res.status(404).json({ message: '团队不存在' });
      }
      
      // 检查用户是否是团队成员
      const member = team.members.find(m => m.user.toString() === userId.toString());
      if (!member) {
        return res.status(403).json({ message: '您不是该团队的成员' });
      }
      
      // 检查权限
      const permissions = getTeamPermissions(member.role);
      if (!permissions.includes(requiredPermission)) {
        return res.status(403).json({ message: '权限不足' });
      }
      
      req.team = team;
      next();
    } catch (error) {
      res.status(500).json({ message: error.message });
    }
  };
};

// 获取团队角色权限
const getTeamPermissions = (role) => {
  const permissions = {
    owner: ['view', 'edit', 'delete', 'manage-members', 'manage-projects'],
    admin: ['view', 'edit', 'manage-members', 'manage-projects'],
    member: ['view', 'edit']
  };
  return permissions[role] || [];
};

// 检查项目权限
const checkProjectPermission = (requiredPermission) => {
  return async (req, res, next) => {
    try {
      const { projectId } = req.params;
      const userId = req.user._id;
      
      const project = await Project.findById(projectId).populate('team');
      if (!project) {
        return res.status(404).json({ message: '项目不存在' });
      }
      
      // 检查项目可见性
      if (project.visibility === 'private') {
        const member = project.members.find(m => m.user.toString() === userId.toString());
        if (!member) {
          return res.status(403).json({ message: '您无权访问此项目' });
        }
        
        const permissions = getProjectPermissions(member.role);
        if (!permissions.includes(requiredPermission)) {
          return res.status(403).json({ message: '权限不足' });
        }
      }
      
      req.project = project;
      next();
    } catch (error) {
      res.status(500).json({ message: error.message });
    }
  };
};

// 获取项目角色权限
const getProjectPermissions = (role) => {
  const permissions = {
    owner: ['view', 'edit', 'delete', 'manage-members', 'manage-tasks'],
    admin: ['view', 'edit', 'manage-members', 'manage-tasks'],
    member: ['view', 'edit']
  };
  return permissions[role] || [];
};

// 检查任务权限
const checkTaskPermission = (requiredPermission) => {
  return async (req, res, next) => {
    try {
      const { taskId } = req.params;
      const userId = req.user._id;
      
      const task = await Task.findById(taskId)
        .populate('createdBy')
        .populate('assignee')
        .populate('team')
        .populate('project');
      
      if (!task) {
        return res.status(404).json({ message: '任务不存在' });
      }
      
      // 任务创建者拥有所有权限
      if (task.createdBy._id.toString() === userId.toString()) {
        req.task = task;
        return next();
      }
      
      // 检查分配者权限
      if (task.assignee && task.assignee._id.toString() === userId.toString()) {
        req.task = task;
        return next();
      }
      
      // 检查协作者权限
      const collaborator = task.collaborators.find(c => 
        c.user.toString() === userId.toString()
      );
      
      if (collaborator && collaborator.permissions.includes(requiredPermission)) {
        req.task = task;
        return next();
      }
      
      // 检查项目权限
      if (task.project) {
        const projectMember = task.project.members.find(m => 
          m.user.toString() === userId.toString()
        );
        
        if (projectMember) {
          const permissions = getProjectPermissions(projectMember.role);
          if (permissions.includes(requiredPermission)) {
            req.task = task;
            return next();
          }
        }
      }
      
      // 检查团队权限
      if (task.team) {
        const teamMember = task.team.members.find(m => 
          m.user.toString() === userId.toString()
        );
        
        if (teamMember) {
          const permissions = getTeamPermissions(teamMember.role);
          if (permissions.includes(requiredPermission)) {
            req.task = task;
            return next();
          }
        }
      }
      
      return res.status(403).json({ message: '权限不足' });
    } catch (error) {
      res.status(500).json({ message: error.message });
    }
  };
};

module.exports = {
  checkTeamPermission,
  checkProjectPermission,
  checkTaskPermission
};

    团队管理接口实现

    // backend/src/routes/teams.js
const express = require('express');
const Team = require('../models/Team');
const auth = require('../middleware/auth');
const { checkTeamPermission } = require('../middleware/permission');

const router = express.Router();

// 创建团队
router.post('/', auth, async (req, res) => {
  try {
    const { name, description } = req.body;
    
    const team = new Team({
      name,
      description,
      owner: req.user._id,
      members: [{
        user: req.user._id,
        role: 'owner'
      }]
    });
    
    await team.save();
    
    res.status(201).json(team);
  } catch (error) {
    res.status(400).json({ message: error.message });
  }
});

// 获取用户所属团队
router.get('/', auth, async (req, res) => {
  try {
    const teams = await Team.find({
      'members.user': req.user._id
    }).populate('owner', 'username');
    
    res.json(teams);
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
});

// 获取团队详情
router.get('/:teamId', auth, checkTeamPermission('view'), async (req, res) => {
  try {
    // 填充成员信息
    const team = await Team.findById(req.params.teamId)
      .populate('members.user', 'username email')
      .populate('owner', 'username');
    
    res.json(team);
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
});

// 更新团队信息
router.put('/:teamId', auth, checkTeamPermission('edit'), async (req, res) => {
  try {
    const { name, description } = req.body;
    
    const team = await Team.findByIdAndUpdate(
      req.params.teamId,
      { name, description, updatedAt: Date.now() },
      { new: true }
    ).populate('members.user', 'username email');
    
    res.json(team);
  } catch (error) {
    res.status(400).json({ message: error.message });
  }
});

// 添加团队成员
router.post('/:teamId/members', auth, checkTeamPermission('manage-members'), async (req, res) => {
  try {
    const { userId, role } = req.body;
    const teamId = req.params.teamId;
    
    const team = await Team.findById(teamId);
    if (!team) {
      return res.status(404).json({ message: '团队不存在' });
    }
    
    // 检查用户是否已经是成员
    const existingMember = team.members.find(m => m.user.toString() === userId);
    if (existingMember) {
      return res.status(400).json({ message: '用户已经是团队成员' });
    }
    
    team.members.push({
      user: userId,
      role: role || 'member'
    });
    
    await team.save();
    
    // 填充用户信息后返回
    const updatedTeam = await Team.findById(teamId)
      .populate('members.user', 'username email');
    
    res.json(updatedTeam);
  } catch (error) {
    res.status(400).json({ message: error.message });
  }
});

// 移除团队成员
router.delete('/:teamId/members/:userId', auth, checkTeamPermission('manage-members'), async (req, res) => {
  try {
    const { teamId, userId } = req.params;
    
    // 不能移除自己（除非是owner）
    if (req.user._id.toString() === userId) {
      return res.status(400).json({ message: '不能移除自己' });
    }
    
    const team = await Team.findById(teamId);
    if (!team) {
      return res.status(404).json({ message: '团队不存在' });
    }
    
    team.members = team.members.filter(m => m.user.toString() !== userId);
    await team.save();
    
    const updatedTeam = await Team.findById(teamId)
      .populate('members.user', 'username email');
    
    res.json(updatedTeam);
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
});

module.exports = router;

    前端实现

    前端通过专门的团队管理界面来支持团队协作功能：

    // frontend/src/pages/TeamManagementPage.tsx
import React, { useEffect, useState } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { fetchTeams, createTeam, addTeamMember } from '../features/teams/teamSlice';

const TeamManagementPage: React.FC = () => {
  const dispatch = useDispatch();
  const { teams, loading, error } = useSelector((state: any) => state.teams);
  
  const [showCreateForm, setShowCreateForm] = useState(false);
  const [teamName, setTeamName] = useState('');
  const [teamDescription, setTeamDescription] = useState('');
  
  useEffect(() => {
    dispatch(fetchTeams());
  }, [dispatch]);
  
  const handleCreateTeam = (e: React.FormEvent) => {
    e.preventDefault();
    dispatch(createTeam({ name: teamName, description: teamDescription }));
    setTeamName('');
    setTeamDescription('');
    setShowCreateForm(false);
  };
  
  if (loading) return <div>加载中...</div>;
  if (error) return <div>错误: {error}</div>;
  
  return (
    <div className="team-management">
      <div className="page-header">
        <h1>团队管理</h1>
        <button 
          className="btn btn-primary"
          onClick={() => setShowCreateForm(!showCreateForm)}
        >
          {showCreateForm ? '取消' : '创建团队'}
        </button>
      </div>
      
      {showCreateForm && (
        <form onSubmit={handleCreateTeam} className="create-team-form">
          <div className="form-group">
            <label>团队名称</label>
            <input
              type="text"
              value={teamName}
              onChange={(e) => setTeamName(e.target.value)}
              required
            />
          </div>
          <div className="form-group">
            <label>团队描述</label>
            <textarea
              value={teamDescription}
              onChange={(e) => setTeamDescription(e.target.value)}
            />
          </div>
          <button type="submit" className="btn btn-success">创建</button>
        </form>
      )}
      
      <div className="teams-list">
        {teams.map((team: any) => (
          <div key={team._id} className="team-card">
            <div className="team-header">
              <h3>{team.name}</h3>
              <span className="team-role">{getRoleDisplayName(team.role)}</span>
            </div>
            <p>{team.description}</p>
            <div className="team-members">
              <h4>成员 ({team.members.length})</h4>
              <div className="members-list">
                {team.members.map((member: any) => (
                  <div key={member.user._id} className="member-item">
                    <span>{member.user.username}</span>
                    <span className="member-role">{getRoleDisplayName(member.role)}</span>
                  </div>
                ))}
              </div>
            </div>
            <div className="team-actions">
              <button className="btn btn-secondary">管理团队</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};

const getRoleDisplayName = (role: string) => {
  const roleMap: Record<string, string> = {
    'owner': '拥有者',
    'admin': '管理员',
    'member': '成员'
  };
  return roleMap[role] || role;
};

export default TeamManagementPage;

    权限控制在任务管理中的应用

    在任务管理界面中，根据用户权限动态显示操作按钮：

    // frontend/src/components/TaskActions.tsx
import React from 'react';

interface TaskActionsProps {
  task: any;
  userRole: string;
  userPermissions: string[];
  onEdit: () => void;
  onDelete: () => void;
  onAssign: () => void;
}

const TaskActions: React.FC<TaskActionsProps> = ({
  task,
  userRole,
  userPermissions,
  onEdit,
  onDelete,
  onAssign
}) => {
  // 检查用户是否有特定权限
  const hasPermission = (permission: string) => {
    return userPermissions.includes(permission);
  };
  
  return (
    <div className="task-actions">
      {hasPermission('edit') && (
        <button className="btn btn-secondary" onClick={onEdit}>
          编辑
        </button>
      )}
      
      {hasPermission('assign') && (
        <button className="btn btn-info" onClick={onAssign}>
          分配
        </button>
      )}
      
      {hasPermission('delete') && (
        <button className="btn btn-danger" onClick={onDelete}>
          删除
        </button>
      )}
      
      {/* 协作者可以添加的额外操作 */}
      <button className="btn btn-outline">添加评论</button>
      <button className="btn btn-outline">添加附件</button>
    </div>
  );
};

export default TaskActions;

    总结

    通过实现完善的团队协作和权限管理系统，todoIng 能够满足不同规模团队的任务管理需求。系统采用多层权限控制机制，既保证了安全性，又提供了足够的灵活性。

    关键特性包括：

    1. 基于角色的权限控制：通过 owner、admin、member 三种角色实现细粒度权限管理
    2. 多层级权限体系：系统级、团队级、项目级、任务级四层权限控制
    3. 灵活的团队管理：支持创建团队、添加成员、分配角色等操作
    4. 任务协作机制：支持任务分配、协作者添加等功能
    5. 前端权限展示：根据用户权限动态显示操作界面

    这套系统使得 todoIng 不仅适用于个人任务管理，也能很好地支持团队协作场景。在下一篇文章中，我们将介绍 todoIng 的 AI 报告生成功能，展示如何利用人工智能技术提升工作效率。
