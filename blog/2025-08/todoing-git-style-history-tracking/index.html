    todoIng Git 风格历史追踪系统实现解析

    在前面的文章中，我们介绍了 todoIng 项目的核心模块实现。今天，我们将深入探讨 todoIng 最具特色的功能——Git 风格的任务历史追踪系统。这个功能使得 todoIng 与其他任务管理工具区别开来，为用户提供了前所未有的任务管理体验。

    为什么需要任务历史追踪？

    在传统的任务管理工具中，我们只能看到任务的当前状态，而无法了解任务是如何一步步演进的。这导致了以下几个问题：

    1. 缺乏上下文：无法了解任务状态变更的原因和背景
    2. 责任不明确：不清楚是谁在什么时候做了什么变更
    3. 无法回溯：当任务状态出现异常时，无法回溯到之前的正确状态
    4. 分析困难：难以分析任务完成的效率和模式

    为了解决这些问题，todoIng 引入了 Git 风格的任务历史追踪系统，为每个任务维护完整的变更历史。

    系统设计思路

    Git 的核心思想是记录每一次变更，形成一个不可变的提交历史。我们借鉴了这一思想，将任务的每一次变更都记录下来，形成任务的历史记录。

    数据模型设计

    任务历史记录模型是整个系统的核心，它记录了任务的每一次变更：

    // backend/src/models/TaskHistory.js
const mongoose = require('mongoose');

const taskHistorySchema = new mongoose.Schema({
  taskId: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'Task',
    required: true
  },
  field: {
    type: String,
    required: true
  },
  oldValue: mongoose.Schema.Types.Mixed,
  newValue: mongoose.Schema.Types.Mixed,
  changedBy: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User',
    required: true
  },
  changeType: {
    type: String,
    enum: ['status-change', 'update', 'assign', 'create', 'snapshot'],
    required: true
  },
  comment: String,
  timestamp: {
    type: Date,
    default: Date.now
  },
  snapshot: mongoose.Schema.Types.Mixed
}, {
  timestamps: true
});

module.exports = mongoose.model('TaskHistory', taskHistorySchema);

    变更类型说明

    1. status-change：任务状态变更（如从"进行中"到"已完成"）
    2. update：任务字段更新（如修改标题、描述等）
    3. assign：任务分配变更
    4. create：任务创建
    5. snapshot：任务快照（记录任务在某个时间点的完整状态）

    实现细节

    自动记录变更

    我们通过 MongoDB 的中间件机制，在任务模型上添加钩子函数，自动记录任务的变更：

    // backend/src/models/Task.js
const mongoose = require('mongoose');
const TaskHistory = require('./TaskHistory');

const taskSchema = new mongoose.Schema({
  title: String,
  description: String,
  status: {
    type: String,
    enum: ['created', 'in-progress', 'paused', 'completed', 'cancelled'],
    default: 'created'
  },
  priority: {
    type: String,
    enum: ['low', 'medium', 'high'],
    default: 'medium'
  },
  assignee: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User'
  },
  createdBy: {
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User',
    required: true
  }
}, {
  timestamps: true
});

// 记录任务创建历史
taskSchema.post('save', function(doc) {
  if (this.isNew) {
    // 记录创建事件
    new TaskHistory({
      taskId: doc._id,
      field: 'task',
      oldValue: null,
      newValue: doc.toJSON(),
      changedBy: doc.createdBy,
      changeType: 'create',
      snapshot: doc.toJSON()
    }).save();
  }
});

// 记录任务更新历史
taskSchema.pre('findOneAndUpdate', async function() {
  const original = await this.model.findOne(this.getQuery());
  this.original = original;
});

taskSchema.post('findOneAndUpdate', async function(doc) {
  if (!this.original) return;
  
  const updated = await this.model.findOne(this.getQuery());
  const changedBy = this.getUpdate().$set?.updatedBy || updated.createdBy;
  
  // 比较变更字段
  const originalData = this.original.toJSON();
  const updatedData = updated.toJSON();
  
  // 记录状态变更
  if (originalData.status !== updatedData.status) {
    new TaskHistory({
      taskId: updated._id,
      field: 'status',
      oldValue: originalData.status,
      newValue: updatedData.status,
      changedBy: changedBy,
      changeType: 'status-change'
    }).save();
  }
  
  // 记录其他字段变更
  Object.keys(updatedData).forEach(key => {
    if (key !== 'status' && 
        key !== '__v' && 
        key !== 'updatedAt' &&
        originalData[key] !== updatedData[key]) {
      
      new TaskHistory({
        taskId: updated._id,
        field: key,
        oldValue: originalData[key],
        newValue: updatedData[key],
        changedBy: changedBy,
        changeType: 'update'
      }).save();
    }
  });
});

module.exports = mongoose.model('Task', taskSchema);

    定期快照机制

    为了能够快速查看任务在某个时间点的完整状态，我们实现了定期快照机制：

    // backend/src/utils/historySnapshot.js
const Task = require('../models/Task');
const TaskHistory = require('../models/TaskHistory');

// 创建任务快照
async function createTaskSnapshot(taskId, userId) {
  try {
    const task = await Task.findById(taskId);
    if (!task) return;

    const snapshot = task.toJSON();
    
    await TaskHistory.create({
      taskId: taskId,
      field: 'snapshot',
      oldValue: null,
      newValue: null,
      changedBy: userId,
      changeType: 'snapshot',
      snapshot: snapshot
    });
  } catch (error) {
    console.error('创建任务快照失败:', error);
  }
}

// 定时创建快照（每天凌晨2点）
function scheduleSnapshots() {
  const cron = require('node-cron');
  
  cron.schedule('0 2 * * *', async () => {
    try {
      // 获取所有活跃任务
      const tasks = await Task.find({
        status: { $in: ['created', 'in-progress', 'paused'] }
      });
      
      // 为每个任务创建快照
      for (const task of tasks) {
        await createTaskSnapshot(task._id, task.createdBy);
      }
      
      console.log(`创建了 ${tasks.length} 个任务快照`);
    } catch (error) {
      console.error('定时快照任务失败:', error);
    }
  });
}

module.exports = {
  createTaskSnapshot,
  scheduleSnapshots
};

    历史查询接口

    为了方便前端展示历史记录，我们提供了专门的历史查询接口：

    // backend/src/routes/taskHistory.js
const express = require('express');
const TaskHistory = require('../models/TaskHistory');
const auth = require('../middleware/auth');

const router = express.Router();

// 获取任务历史记录
router.get('/task/:taskId', auth, async (req, res) => {
  try {
    const { taskId } = req.params;
    const { limit = 50, offset = 0 } = req.query;
    
    // 验证用户权限
    // 这里需要检查用户是否有权查看该任务的历史记录
    
    const history = await TaskHistory.find({ taskId })
      .sort({ timestamp: -1 })
      .limit(parseInt(limit))
      .skip(parseInt(offset))
      .populate('changedBy', 'username');
    
    const totalCount = await TaskHistory.countDocuments({ taskId });
    
    res.json({
      history,
      totalCount,
      hasMore: totalCount > parseInt(offset) + parseInt(limit)
    });
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
});

// 获取时间线视图
router.get('/timeline', auth, async (req, res) => {
  try {
    const { startDate, endDate } = req.query;
    
    const query = {
      changedBy: req.user._id,
      timestamp: {
        $gte: startDate ? new Date(startDate) : new Date(Date.now() - 30 * 24 * 60 * 60 * 1000),
        $lte: endDate ? new Date(endDate) : new Date()
      }
    };
    
    const timeline = await TaskHistory.find(query)
      .sort({ timestamp: 1 })
      .populate('taskId', 'title')
      .populate('changedBy', 'username');
    
    res.json(timeline);
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
});

module.exports = router;

    前端展示实现

    前端通过时间线组件展示任务的历史记录：

    // frontend/src/components/TaskHistoryTimeline.tsx
import React, { useEffect, useState } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { fetchTaskHistory } from '../features/tasks/taskSlice';

interface HistoryItem {
  _id: string;
  taskId: string;
  field: string;
  oldValue: any;
  newValue: any;
  changedBy: {
    username: string;
  };
  changeType: string;
  comment?: string;
  timestamp: string;
}

const TaskHistoryTimeline: React.FC<{ taskId: string }> = ({ taskId }) => {
  const dispatch = useDispatch();
  const [history, setHistory] = useState<HistoryItem[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const loadHistory = async () => {
      setLoading(true);
      // 这里调用 dispatch(fetchTaskHistory(taskId))
      // 为了简化示例，我们直接模拟数据
      setLoading(false);
    };

    if (taskId) {
      loadHistory();
    }
  }, [taskId, dispatch]);

  const formatChange = (item: HistoryItem) => {
    switch (item.changeType) {
      case 'create':
        return `创建了任务`;
      case 'status-change':
        return `将状态从 "${item.oldValue}" 变更为 "${item.newValue}"`;
      case 'update':
        return `更新了 ${item.field} 字段`;
      case 'assign':
        return `将任务分配给 ${item.newValue?.username || '未分配'}`;
      case 'snapshot':
        return `创建了任务快照`;
      default:
        return `进行了未知变更`;
    }
  };

  if (loading) {
    return <div>加载历史记录中...</div>;
  }

  return (
    <div className="history-timeline">
      <h3>任务历史</h3>
      <div className="timeline">
        {history.map((item) => (
          <div key={item._id} className="timeline-item">
            <div className="timeline-marker"></div>
            <div className="timeline-content">
              <div className="timeline-header">
                <span className="timeline-user">{item.changedBy.username}</span>
                <span className="timeline-time">
                  {new Date(item.timestamp).toLocaleString()}
                </span>
              </div>
              <div className="timeline-description">
                {formatChange(item)}
              </div>
              {item.comment && (
                <div className="timeline-comment">
                  备注: {item.comment}
                </div>
              )}
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};

export default TaskHistoryTimeline;

    性能优化

    考虑到历史记录可能会快速增长，我们采取了以下性能优化措施：

    1. 索引优化：为常用查询字段创建索引
    2. 分页查询：限制单次查询返回的记录数量
    3. 数据归档：定期将历史数据归档到冷存储
    4. 缓存机制：对频繁访问的历史数据进行缓存
    // backend/src/models/TaskHistory.js
// 添加索引
taskHistorySchema.index({ taskId: 1, timestamp: -1 });
taskHistorySchema.index({ changedBy: 1, timestamp: -1 });
taskHistorySchema.index({ changeType: 1 });
